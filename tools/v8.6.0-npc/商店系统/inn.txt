lhz_in02,230,284,4	script	饭店 职员#01	86,{
	mes "[饭店 职员]";
	mes "欢迎光临";
	mes "里希塔乐镇,";
	mes "这里可以找到";
	mes "最好的住宿";
	mes "与最好的服务.";
	next;
	switch(select("储存重生点:休息 - 5,000 zeny:取消")) {
	case 1:
		savepoint "lhz_in02",209,275;
		mes "[饭店 职员]";
		mes "感谢你,";
		mes "你的重生点";
		mes "已储存在";
		mes "这里.";
		close;
	case 2:
		if(Zeny < 5000) {
			mes "[饭店 职员]";
			mes "很抱歉,";
			mes "你似乎不足 5,000 zeny";
			mes "请在确认一下是否有足够的金钱.";
			close;
		}
		set Zeny,Zeny - 5000;
		mes "[饭店 职员]";
		mes "谢谢您~";
		mes "希望您喜欢这里";
		mes "能在里希塔乐镇";
		mes "多停留一会.";
		close2;
		warp "lhz_in02",219,150;
		percentheal 100,100;
		npcskill "AL_BLESSING",10,99,99;
		end;
	case 3:
		mes "[饭店 职员]";
		mes "谢谢你并祝您";
		mes "有个愉快的一天.";
		close;
	}
}

//======================= Inn Function ==============================
function	script	F_InnMaid	{
	mes getarg(0);
	mes "欢迎光临";
	mes getarg(1) + ".";
	mes "有什么需要我帮忙的呢?";
	next;
	switch(select("储存点:休息一会 -> 5000 zeny:取消")) {
	case 1:
		mes getarg(0);
		mes "你的重生点";
		mes "已储存.";
		mes "感谢您,";
		mes "欢迎再次光临.";
		savepoint getarg(2),getarg(3),getarg(4);
		close;
	case 2:
		mes getarg(0);
		if(Zeny < 5000){
			mes "很抱歉, 服务费用是 5,000 zeny. 请确认你有足够的金钱在过来, 好吗?";
			close;
		}
		mes "感谢您.";
		mes "希望您能";
		mes "好好休息~";
		close2;
		set Zeny,Zeny - 5000;
		percentheal 100,100;
		return;
	case 3:
		close;
	}
}
