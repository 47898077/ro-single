hugel,178,142,0	script	飞空艇入口#hugel	45,1,1,{
	end;
OnTouch:
	mes "如果要搭乘飞空艇的话，需要 1,200 zeny来支付飞机票的费用呢.";
	mes "请问是否愿意搭乘飞空艇呢?";
	next;
	if (select("是的:不要") == 1) {
		if (countitem(7311) > 0) {
			delitem 7311,1; //Free_Flying_Ship_Ticket
			warp "airplane",244,58;
			end;
		}
		if (Zeny >= 1200) {
			set Zeny, Zeny - 1200;
			warp "airplane",244,58;
			end;
		}
		mes "抱歉,您身上所带的zeny不够.";
		mes "请记住我们的费用是 1,200 zeny.";
		close;
	}
	mes "谢谢您选择了我们的服务，欢迎下次再来.";
	close;
}
