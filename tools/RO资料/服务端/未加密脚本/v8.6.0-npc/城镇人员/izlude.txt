-	script	::Sailor_izlude	-1,{
	mes "[船员]";
	mes "各位, 注意了!";
	mes "这艘拥有超高航速的船早已准备好随时出发,";
	mes "不过它不能保证大家的安全就是了！";
	mes "来吧，我们走!";
	mes "快点啊, 别磨磨蹭蹭的!";
	next;
	switch(select("柏伊亚岚岛 -> 150 Zeny.:艾尔贝塔码头  -> 500 Zeny.:取消.")) {
	case 1:
		if (Zeny < 150) {
			mes "[船员]";
			mes "150 Zeny!";
			mes "仅仅需要 150 Zeny 即可搭乘!";
			close;
		}
		set Zeny, Zeny - 150;
		warp "izlu2dun",107,50;
		end;
	case 2:
		if (Zeny < 500) {
			mes "[船员]";
			mes "500 Zeny!";
			mes "仅仅需要 500 Zeny 即可搭乘!";
			close;
		}
		set Zeny, Zeny - 500;
		warp "alberta",188,169;
		end;
	case 3:
		close;
	}
}
