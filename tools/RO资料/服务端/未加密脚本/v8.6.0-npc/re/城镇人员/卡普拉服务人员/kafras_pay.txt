// Function: Sets variables for Payon Kafras ---------------------------------
function	script	F_KafSetPay	{

	setarray @wrpD$[0], "艾尔贝塔", "普隆德拉", "梦罗克";

	set @wrpC$[0], @wrpD$[0]+" -> "+@wrpP[0];

	set @wrpC$[2], @wrpD$[2]+" -> "+@wrpP[2];
	set @wrpC$[3], "取消";
	return;
}
