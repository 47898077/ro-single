	mes "我可以为你锻治装置项目中的物品。";

	set .@refineitemid, getequipid(.@part); // save id of the item
	set .@refinerycnt, getequiprefinerycnt(.@part); //save refinery count
	switch(getequipweaponlv(.@part)){
	case 0: 	//Refine Armor
		set .@price,2000;
		set .@material,985;
		set .@safe,4;
		break;
	case 1: 	//Refine Level 1 Weapon
		set .@price,50;
		set .@material,1010;
		set .@safe,7; 
		break;
	case 2: 	//Refine Level 2 Weapon
		set .@price,200;
		set .@material,1011;
		set .@safe,6; 
		break;
	case 3: 	//Refine Level 3 Weapon
		set .@price,5000;
		set .@material,984;
		set .@safe,5; 
		break;
	case 4: 	//Refine Level 4 Weapon
		set .@price,20000;
		set .@material,984;
		set .@safe,4; 
		break;
	case 5: 	//Refine other stuff?
		set .@price,2000;
		set .@material,985;
		set .@safe,4; 
		break;
	}
	if(.@features != 1) {
		mes "[" + getarg(0) + "]";
		mes "精练这个物品我需要";
		mes "1件 ^003366"+getitemname(.@material)+"^000000 和";
		mes "服务费 " + .@price + " Zeny.";

		next;

			mes "[" + getarg(0) + "]";
			mes "耶...";
			mes "那好吧";
			mes "欢迎下次再来";
			close;

			mes "我要继续精炼他了，";

			case 985:
				mes "看来这武器已经锻治很多次了的样子...要是再继续锻治的话，可能会造成损坏喔!000000!!";

				mes "要是损坏的话可就不能继续使用了!!插在里面的卡片和特征也^ff0000全部都会消失^000000!!";

			mes "我要再说清楚一次";

			close;

			emotion e_an;
			mes "等一会儿..";
			mes "你以为我是笨蛋吗?!";
			mes "快出去,不然我会用我的铁锤打你..";
			close;

				mes "非常对不起...";

// New Refining Functions ========================
	if(getequiprefinerycnt(.@part) < .@safe) {

			if(select("好的...","算了...") == 2){
				mes "[" + getarg(0) + "]";
				mes "这可是你说的哦...";
				close;
			}
		}
		break;
	case 3:
		next;
		mes "[" + getarg(0) + "]";
		mes "这可是你说的哦...";
		close;
	}
	set .@fullprice,.@price * .@refinecnt;
	mes "[" + getarg(0) + "]";
	mes "总共需要" + .@refinecnt + " " + getitemname(.@material) + " 以及费用 " + .@fullprice + " Zeny... 要锻冶吗?";
	next;
	if(select("好的","算了...") == 2){
		mes "[" + getarg(0) + "]";
		mes "这可是你说的哦...";
		close;
	}
	if(countitem(.@material) < .@refinecnt || Zeny < .@fullprice) {
		mes "[" + getarg(0) + "]";
		mes "这些就是你所拥有的所有物品吗?虽然很可惜，但是你的材料不足，我也实在爱莫能助，再说，我付出也应该得要有所收获嘛!";
		close;
	}
	set Zeny,Zeny - .@fullprice;
	delitem .@material,.@refinecnt;
	while(.@refinecnt){
		if (getequipisequiped(.@part) == 0) {
			mes "[" + getarg(0) + "]";
			mes "对不起,材料不够我也不能帮你.....";
			close;
		}
		if (getequipid(.@part) != .@refineitemid || (.@menu2 == 1 && getequippercentrefinery(.@part) < 100)) {
			mes "[" + getarg(0) + "]";
			mes "不要开玩笑了,你以为我是笨蛋吗 !! !?!";
			mes "快出去...";
			mes "不然我会用我的铁锤打你..!!!";
			close;
		} 
		mes "匡匡, 匡匡!!!";
		if(.@menu2 == 2 && getequippercentrefinery(.@part) <= rand(100)) {
			failedrefitem .@part;
			emotion e_omg;
			mes "[" + getarg(0) + "]";
			mes "咳咳咳咳咳咳咳!!! 武器在锻治的过程中损坏了...";
			set .@refinecnt,.@refinecnt - 1;
			if(.@refinecnt == 0) close;
			mes "这是剩下的钱和材料...";
			getitem .@material,.@refinecnt;
			set .@fullprice,.@refinecnt * .@price;
			set Zeny,Zeny + .@fullprice;
			close;
		}
		successrefitem .@part;
		emotion e_no1;
		set .@refinecnt,.@refinecnt - 1;
		next;
	}
	mes "[" + getarg(0) + "]";
	mes "所有的都完成了，欢迎下次再来.";
	close;		
}

// Equipment Repair Function
//============================================================
function	script	repairmain	{
	set .@repairprice,5000;
	mes "["+getarg(0)+"]";
	mes "你好!";
	mes "你想让我修理东西吗?";
	mes "我可以帮你修理坏掉的装备及武器!";
	next;
	switch(select("恩...有东西修...:目前没有")) {
	case 1:
		set .@checkitem,1;
		while (1) {
			if (getbrokenid(.@checkitem) == 0) {
				break;
			}
			set .@checkitem,.@checkitem+1;
		}
		set .@checkitem,.@checkitem-1;
		if (!.@checkitem) {
			mes "["+getarg(0)+"]";
			mes "噢!这真是难以置信";
			mes "你肯定非常关心你的物品!没有一件损坏了哦";
			next;
			mes "["+getarg(0)+"]";
			mes "如果每个人都像你这样，我就会要失业了!";
			close;
		}
		mes "["+getarg(0)+"]";
		mes "呃...";
		mes "让我看看...";
		mes "你的物品中,";
		mes "" + .@checkitem + " are damaged.";
		mes "你想修理他们吗?";
		next;
		set .@totalcost,.@repairprice*.@checkitem;
		mes "["+getarg(0)+"]";
		mes "修理一件需要 " + .@repairprice + " Zeny. 所以修理你全部损坏的物品需要 " + .@totalcost + " Zeny! 你想修理他们吗?";
		next;
		switch(select("Yes:No")) {
		case 1:
			if (Zeny < .@totalcost) {
				mes "["+getarg(0)+"]";
				mes "呃呃..";
				mes "你的金钱不足.";
				close;
			}
			set .@checkitem2,1;
			while (1) {
				if (getbrokenid(.@checkitem2) == 0) {
					break;
				}
				set .@checkitem2,.@checkitem2+1;
			}
			set .@checkitem2,.@checkitem2-1;
			if (.@checkitem == .@checkitem2) {
				set zeny,zeny-.@totalcost;
				while (.@checkitem) {
					repair(.@checkitem);
					set .@checkitem,.@checkitem-1;
				}
				mes "["+getarg(0)+"]";
				mes "Ok! 修好了。";
				close;
			}
			else {
				mes "["+getarg(0)+"]";
				mes "看来你并没有任何坏掉的东西需要修理呐...";
				close;
			}
		case 2:
			mes "["+getarg(0)+"]";
			mes "我可以帮你修理坏掉的装备及武器";
			close;
		}
	case 2:
		mes "["+getarg(0)+"]";
		mes "哦...";
		mes "你没有";
		mes "工作需要我来做";
		mes "如果你没有任何东西";
		mes "需要修理.";
		close;
	}
}
