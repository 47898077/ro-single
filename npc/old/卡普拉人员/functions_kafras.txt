//=== Main Function ========================================================
//=   arg(0): Used to determine which welcome message to show.
//=   arg(1): Used to determine which menu to display.
//=   arg(2): Used to determine if the info menu is shown in F_KafInfo.
//=   arg(3): Cost to use Storage
//=   arg(4): Cost to Rent a Pushcart
//==========================================================================
function	script	F_Kafra	{
	callfunc "F_ClearGarbage"; // Clear outdated, unused variables
	// Display Kafra Welcome Message
	switch(getarg(0)){
		default:
		case 0:
			// Default message (obsolete)
			mes "[卡普拉雇员]";
			mes "---------------------------";
			mes "欢迎来到";
			mes "卡普拉服务中心.";
			mes "卡普拉的服务";
			mes "永远在您身边.有什么可以帮助您的吗?";
			break;
		case 1:
			// Niflheim specific message
			mes "[卡普拉雇员]";
			mes "^666666W-weeeelc-c-come";
			mes "to th-the K-kaaafrrrra";
			mes "C-coorpoor-r-ratioooonn...^000000";
			break;
		case 2:
			// Guild Castle Kafra message
			set @GID,getcharid(2);
			mes "[卡普拉雇员]";
	 		mes "欢迎光临. ^ff0000" + GetGuildName(@GID) + "^000000 会员.";
			mes "卡普拉服务中心可以带您去任何您想去的地方.";
			break;
		case 3:
			// Amatsu specific message (obsolete)
			mes "[卡普拉雇员]";
			mes "所以，您来自一个遥远的土地来研究我们的文化，或者您只是观光?";
			mes "在这种情况下, 为什么不停留片刻?";
			mes "空气中充满了";
			mes "永恒的沉重与愉快的野花的香味.";
			break;		
		case 4:
			// Louyang and Ayothaya specific message (obsolete)
			mes "[卡普拉雇员]";
			mes "与我们许多的卡普拉服务人员";
			mes "服务区域, 您从未";
			mes "离开这个家.";
			break;		
		case 5:
			// NPC has it's own welcome message. (Display nothing)
			break;
	}
	next;
	while (1) {
		deletearray @K_Menu0$[0],getarraysize(@K_Menu0$);
		if (getarg(0) == 2) {
			// Guild Kafra Menu override (free Teleport, free Storage, Free Cart)
			setarray @K_Menu0$[0],"- 使用仓库服务","- 使用公会仓库","- 手推车租赁服务","- 使用传送服务","- 取消";
		} else {
			switch(getarg(1)){
				// Save and Storage only
				case 1:	setarray @K_Menu0$[0],"- 存储服务","- 使用仓库服务","- 取消"; break;
				// Storage only
				case 2:	setarray @K_Menu0$[0],"- 使用仓库服务","- 取消"; break;
				// No Teleport (Common)
				case 3: setarray @K_Menu0$[0],"- 存储服务","- 使用仓库服务","- 手推车租赁服务","- 查询其它情报信息","- 取消"; break;
				// Case 4 is Einbroch no tele message.
				// No save, or teleport. (Common)
				case 5:	setarray @K_Menu0$[0],"- 使用仓库服务","- 手推车租赁服务","- 查询其它情报信息","- 取消"; break;
				// Storage and Check Other Information only.
				case 6:	setarray @K_Menu0$[0],"- 使用仓库服务","- 查询其它情报信息","- 取消"; break;
				// Save, Storage, and Pushcart only (Kafra Warehouse)
				case 7:	setarray @K_Menu0$[0],"- 存储服务","- 使用仓库服务","- 手推车租赁服务","- 取消"; break;
				// Save, Storage, Other Check information. (Turbo track)
				case 8: setarray @K_Menu0$[0],"- 存储服务","- 使用仓库服务","- 查询其它情报信息","- 取消"; break;
				// No Save (Rune Knight)
				case 9: setarray @K_Menu0$[0],"- 使用仓库服务","- 手推车租赁服务","- 使用传送服务","- 查询其它情报信息","- 取消"; break;
				// Storage, Save, and Pushcart (Dewata, reorder of case 7)
				case 10: setarray @K_Menu0$[0],"- 使用仓库服务","- 存储服务","- 手推车租赁服务","- 取消"; break;
				// Default message (obsolete)
				default: setarray @K_Menu0$[0],"- 存储服务","- 使用仓库服务","- 使用传送服务","- 手推车租赁服务","- 查询其它情报信息","- 取消"; break;
			}
		}
		set .@menu$,"";
		for (set .@i, 0; .@i < getarraysize(@K_Menu0$); set .@i, .@i + 1)
			set .@menu$, .@menu$ + @K_Menu0$[.@i] + ":";	
		set .@j, select (.@menu$) - 1;
		if (@K_Menu0$[.@j] == "- 存储服务") {
			return;
		} else if (@K_Menu0$[.@j] == "- 使用仓库服务") {
			// Do not charge for Guild Storage
			if(getarg(0) == 2) 
				callfunc "F_KafStor",2,0,0;
			else 
				callfunc "F_KafStor",0,getarg(3),getarg(0);
			next;
		} else if (@K_Menu0$[.@j] == "- 使用传送服务") {
			// Display Einbroch "没有传送服务" notice.
			if(getarg(1) != 4) callfunc "F_KafTele",getarg(0);
			else {
				mes "[卡普拉雇员]";
				mes "---------------------------";
				mes "因为 ^FF0000被限制";
				mes "根据传送协定^000000, ";
				mes "卡普拉服务中心无法";
				mes "提供传送服务在秀发兹发德教国中.";
				next;
				mes "[卡普拉雇员]";
				mes "我们想请您用";
				mes "飞艇服务替代";
				mes "感谢您的";
				mes "理解和支持.";
			}
			next;
		} else if (@K_Menu0$[.@j] == "- 手推车租赁服务") {
			if(callfunc("F_KafCart",getarg(0),getarg(4)) == 1) 
				next;
		} else if (@K_Menu0$[.@j] == "- 查询其它情报信息") {
			callfunc "F_KafInfo",getarg(2);
			next;
		} else if (@K_Menu0$[.@j] == "- 取消"){
			callfunc "F_KafEnd",getarg(0),0;
			end;
		} else if (@K_Menu0$[.@j] == "- 打开公会仓库") {
			callfunc "F_KafStor",1,0;
			next;
		}
	}
}

//===================================================
// Storage Function
//===================================================
function	script	F_KafStor	{
	// Unable to access Guild Storage (Busy)
	if(getarg(0) == 1){
		if(guildopenstorage()){
			mes "[卡普拉雇员]";
			mes "对不起，公会有另外一位成员正在使用公会仓库";
			mes "现在. 请等待另一位成员用完.";
			close2;
			cutin "", 255;
			end;
		}
		cutin "", 255;
		close;
	}
	// Unable to access Normal Storage (Insufficient Basic Skills)
	if(basicskillcheck() && getskilllv("NV_BASIC") < 6){
		mes "[卡普拉雇员]";
		// Niflheim Specific Message
		if (getarg(2) == 1) {
			mes "^666666对…对……对不起,";
			mes "您…您是一个";
			mes "初心者... 需…需要";
			mes "基本职业技能等级 l-等级 6...^000000";
			return;
		}
		mes "对不起, 您";
		mes "需要基本职业";
		mes "技能等级 6才能";
		mes "使用仓库服务.";
		return;
	}
	// Accessing Normal Storage (Skipped if accessing Storage from Guild castle)
	if(getarg(0) != 2){
		// Consume "卡普拉仓库免费使用券" if available.
		if(countitem(7059)) delitem 7059,1;
		else {
			if(Zeny<getarg(1)){
				mes "[卡普拉雇员]";
				// Niflheim Specific Message
				if (getarg(2) == 1) {
					percentheal -50,-50;
					mes "^666666zeny...";
					mes "您需要更多的zeny...!";
					mes "需要 150... zeny...";
					mes "Ergh! T-taking bl-blood~!^000000";
					return;
				}
				// Standard Message
				mes "对不起,但是您";
				mes "没有足够的 zeny 去使用";
				mes "仓库服务. 我们";
				mes "仓库使用费用需要 "+getarg(1)+" zeny.";
				return;
			}
			set Zeny, Zeny-getarg(1);
			set RESRVPTS, RESRVPTS + (getarg(1)/5);
		}
	}

	// Surpress the "End" message if in guild castle.
	if (getarg(0) != 2) {
		// Niflheim specific Message
		if (getarg(2) == 1) {
			percentheal 0,-10;
			mes "[卡普拉雇员]";
			for (set .@i, 0; .@i < 5; set .@i, .@i + 1)
				mes "^666666感谢您.. 的... 使用...^000000";
		}
		// Normal message
		else {
			mes "[卡普拉雇员]";
			mes "在这里，让我打开它";
			mes "您的仓库.";
			mes "感谢您使用";
			mes "卡普拉服务.";
		}
	}
	callfunc("F_CheckKafCode");	//check your storage password, if set
	close2;
	openstorage;
	cutin "", 255;
	end;
}

function	script	F_KafTele	{
	mes "[卡普拉雇员]";

//===================================================
// Cart Function
//===================================================
function	script	F_KafCart	{
	// Ensure that the class wanting to rent a pushcart is a merchant
	if(BaseClass != Job_Merchant){
		mes "[卡普拉雇员]";
		mes "对不起, 但是";
		mes "手推车租赁服务";
		mes "是只开放给商人系列职业,";
		mes "例如银匠, 炼金术士等.";
		mes "和超级初心者的.";
		return 1;
	}
	// Make sure the invoking character does not have a cart already
	else if(checkcart() == 1){
		mes "[卡普拉雇员]";
		mes "您已经拥有了";
		mes "一辆手推车.";
		mes "不幸的是, 我们不能";
		mes "在同一时间";
		mes "不能租赁更多的给顾客.";
		return 1;
	}
	// Consume "卡普拉手推车服务免费使用券" if available.
	// Do not consume if in Guild castle.
	if(countitem(7061) > 0 && getarg(0) != 2) delitem 7061,1;
	else {
		mes "[卡普拉雇员]";
		mes "手推车租赁, 的费用需要.";
		mes ""+getarg(1)+" zeny. ";
		mes "您想租一辆手推车吗?";
		next;
		if (select("- 租一辆手推车.:- 取消") == 2) return 0;
			if(Zeny<getarg(1)){
				mes "[卡普拉雇员]";
				mes "对不起, 但是您";
				mes "没有足够的";
				mes "zeny 去支付租赁手推车";
				mes "所需要的 "+getarg(1)+" zeny.";
				return 1;
			}
			set Zeny,Zeny-getarg(1);
			set RESRVPTS, RESRVPTS + 48;
	}
	setcart;
	return 1;
}

//===================================================
// Check Storage Password Function
//===================================================
function	script	F_CheckKafCode	{
	if(!#kafra_code) return;
	mes "请输入您的仓库密码:";
	set @code_,0;
	input @code_;
	if(@code_ != #kafra_code) {
		dispbottom "密码错误.";
		close2;
		cutin "",255;
		end;
	}
	set @kafcode_try,0;
	set @code_,0;
	return;
}

// === Set / Change / Clear Storage Password Function ===
// getarg(0): NPC Name
// getarg(1): Company Name
//=======================================================
function	script	F_SetKafCode	{
	mes getarg(0);
	if(#kafra_code) {
		mes "您的仓库现在有密码保护了.您现在还想做什么?";
		next;
		switch (select("- 修改密码 -> 5000z:- 删除仓库密码 -> 1000z:- 取消")) {
			Case 1:
				mes getarg(0);
				mes "首先, 请您输入 ^0000FF旧密码^000000.";
				set @code,callfunc("F_EntKafCode");
				if(!@code || @code != #kafra_code) {
					mes "密码错误. 您无法修改密码，请重新操作.";
					emotion e_hmm;
					break;
				}
				next;
				callsub S_SET,getarg(0),getarg(1);
				break;
			Case 2:
				mes getarg(0);
				mes "请输入您的仓库密码.";
				set @code,callfunc("F_EntKafCode");
				if(!@code) {
					mes "密码未被删除.";
					emotion e_hmm;
					break;
				}
				next;
				mes getarg(0);
				if(Zeny < 1000) {
					mes "您没有足够的zeny.";
					emotion e_cash;
					break;
				}
				set Zeny,Zeny-1000;
				if(@code == #kafra_code) {
					set #kafra_code,0;
					mes "您已成功清除了您的旧密码.";
					mes "感谢您使用 "+getarg(1)+".";
					emotion e_thx;
				} else {
					mes "密码错误. 我们不会归还您的 1000z.";
					mes "请输入正确的密码，再试一次.";
					emotion e_sry;
				}
				break;
			Case 3:
				break;
		}
	} else {
		mes getarg(1)+" 自豪地介绍您一个新的服务:";
		mes "为您的仓库增加密码保护.";
		next;
		switch (select("- 请设定一个新密码 -> 5000z:- 取消")) {
			Case 1: callsub S_SET,getarg(0),getarg(1); break;
			Case 2: break;		
		}
	}
	close2;
	cutin "",255;
	end;

S_SET:
	mes getarg(0);
	mes "请输入 ^FF0000新密码^000000 以防止仓库被盗.";
	set @code,callfunc("F_EntKafCode");
	if(!@code) {
		mes "密码没有改变.";
		emotion e_hmm;
		return;
	}
	next;
	mes getarg(0);
	if(Zeny < 5000) {
		mes "您没有足够的Zeny.";
		emotion e_cash;
		return;
	}
	set Zeny,Zeny-5000;
	set #kafra_code,@code;
	mes "您已经拥有了一个密码来保护您的仓库.";
	mes "感谢您的使用 "+getarg(1)+".";
	emotion e_thx;
	return;
}

//===================================================
// Basic Password Validation Function
//===================================================
function	script	F_EntKafCode	{
	mes "请输入一个1000~10000000之间的数字:";
	set @code_,0;
	set @kafcode_try,@kafcode_try+1;
	if(@kafcode_try>10) {
		set @kafcode_try,0;
		logmes "破解: 尝试以符合仓库密码.";
	}
	if(input(@code_) == 1) {
		mes "您的密码太长.";
		return 0;
	}
	if(@code_<1000) {
		mes "您的密码太短.";
		return 0;
	}
	return @code_;
}
